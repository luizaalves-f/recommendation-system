# Em outra janela do PowerShell como administrador:
cd "C:\Users\luiiz\OneDrive\Documentos\GitHub\recommendation-system"

# Testar endpoint inicial:
Invoke-RestMethod -Uri "http://127.0.0.1:5000/" -Method Get

# Listar generos vindos do banco SQLite:
Invoke-RestMethod -Uri "http://127.0.0.1:5000/generos" -Method Get

# Testar recomendacao por genero existente no banco.
# Este endpoint agora consulta o recsys.db, nao o arquivo regras_recomendacao.json.
$body = @{ genero = "Action" } | ConvertTo-Json
Invoke-RestMethod `
  -Uri "http://127.0.0.1:5000/recomendar" `
  -Method Post `
  -Body $body `
  -ContentType "application/json"

# Testar outro genero usando o nome como esta no dataset:
$body = @{ genero = "Science Fiction" } | ConvertTo-Json
Invoke-RestMethod `
  -Uri "http://127.0.0.1:5000/recomendar" `
  -Method Post `
  -Body $body `
  -ContentType "application/json"

# Testar erro de validacao quando o corpo nao tem genero:
$body = @{} | ConvertTo-Json
Invoke-RestMethod `
  -Uri "http://127.0.0.1:5000/recomendar" `
  -Method Post `
  -Body $body `
  -ContentType "application/json"

# Testar genero inexistente:
$body = @{ genero = "genero inexistente" } | ConvertTo-Json
Invoke-RestMethod `
  -Uri "http://127.0.0.1:5000/recomendar" `
  -Method Post `
  -Body $body `
  -ContentType "application/json"
